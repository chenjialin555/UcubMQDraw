## 一、风格迁移 `tools/style-transfer/tool.ts`

```ts
import StyleTransferForm from './StyleTransferForm.vue'
import StyleTransferPreview from './StyleTransferPreview.vue'

export const styleTransferTool = {
  key: 'style_transfer',
  name: '风格迁移',
  description: '上传底图和风格参考图，生成指定风格图片',

  Form: StyleTransferForm,
  Preview: StyleTransferPreview,

  createForm() {
    return {
      baseImageUrl: '',
      styleImageUrl: '',
      generationCount: 1,
      styleStrength: 0.75,
      description: '',
    }
  },

  validate(form: any) {
    if (!form.baseImageUrl) return '请上传底图'
    if (!form.styleImageUrl) return '请上传风格参考图'
    if (form.description.length > 300) return '描述不能超过 300 字'
    return ''
  },

  buildRequest(form: any) {
    return {
      toolKey: 'style_transfer',
      params: {
        prompt: form.description || '风格迁移',
        refImageList: [form.baseImageUrl, form.styleImageUrl],
        generationCount: form.generationCount,
        styleStrength: form.styleStrength,
      },
    }
  },
}
```

## 二、局部重绘 `tools/inpaint/tool.ts`

```ts

```

## 三、工具列表 `tools/index.ts`

```ts
import { styleTransferTool } from './style-transfer/tool'
import { inpaintTool } from './inpaint/tool'

export const tools = [styleTransferTool, inpaintTool]

export function getTool(key: string) {
  return tools.find(t => t.key === key)
}

export function getDefaultTool() {
  return tools[0]
}
```
