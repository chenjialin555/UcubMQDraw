# 后端工具处理器示例

> 与前端 `toolKey` 一一对应。**下发 JSON 字段以 [mq_contract.md](../mq_contract.md) 为准**，无 `workflow` 参数。

## 一、风格迁移 `tools/style_transfer.py`

```python
class StyleTransferTool:
    key = "style_transfer"
    name = "风格迁移"

    def validate(self, params: dict) -> None:
        ref_images = params.get("refImageList") or []
        if len(ref_images) < 2:
            raise ValueError("风格迁移需要底图和风格参考图")

    def build_task_dispatch_params(
        self,
        task_id: str,
        inner_task_id: str,
        params: dict,
    ) -> dict:
        # task_id 仅存 DB，不进 MQ；MQ 只用 innerTaskId
        return {
            "innerTaskId": inner_task_id,
            "modelName": params.get("modelName", "SDXL/Flux"),
            "prompt": params.get("prompt") or self.name,
            "width": params.get("width", 1024),
            "height": params.get("height", 1024),
            "batchSize": params.get("generationCount", 1),
            "refImageList": params.get("refImageList") or [],
            "timeout": params.get("timeout", 300),
            "styleStrength": params.get("styleStrength", 0.75),
        }
```

## 二、局部重绘 `tools/inpaint.py`

```python
class InpaintTool:
    key = "inpaint"
    name = "局部重绘"

    def validate(self, params: dict) -> None:
        ref_images = params.get("refImageList") or []
        if not ref_images:
            raise ValueError("局部重绘需要底图")
        if not params.get("maskImageUrl"):
            raise ValueError("局部重绘需要蒙版图")

    def build_task_dispatch_params(
        self,
        task_id: str,
        inner_task_id: str,
        params: dict,
    ) -> dict:
        return {
            "innerTaskId": inner_task_id,
            "modelName": params.get("modelName", "SDXL/Flux"),
            "prompt": params.get("prompt") or self.name,
            "width": params.get("width", 1024),
            "height": params.get("height", 1024),
            "batchSize": params.get("batchSize", 1),
            "refImageList": params.get("refImageList") or [],
            "maskOssUrl": params.get("maskImageUrl"),
            "timeout": params.get("timeout", 300),
            "denoiseStrength": params.get("denoiseStrength", 0.65),
            "maskBlurRadius": params.get("maskBlurRadius", 8),
        }
```

## 三、注册 `tools/registry.py`

```python
from app.tools.style_transfer import StyleTransferTool
from app.tools.inpaint import InpaintTool

tools = {
    "style_transfer": StyleTransferTool(),
    "inpaint": InpaintTool(),
}

def get_tool(tool_key: str):
    tool = tools.get(tool_key)
    if not tool:
        raise ValueError(f"不支持的工具类型: {tool_key}")
    return tool
```

## 四、TaskService 调用方式（一般不用改）

```python
tool = get_tool(tool_key)
tool.validate(params)

task_dispatch_params = tool.build_task_dispatch_params(
    task_id=task_id,
    inner_task_id=inner_task_id,
    params=params,
)

# DB 存 task_id + inner_task_id + task_dispatch_params
# 创建任务后始终写 task_dispatch_params 并投递 RocketMQ。
# 本地联调用 imggen-stub 消费 task topic 并回调 callback topic。
mq_service.send(task_dispatch_params, tool_key=tool.key)
```
