## 一、状态与前端展示映射

```ts
// frontend/src/utils/taskStatus.ts

export const STATUS_LABEL: Record<string, string> = {
  created: '已创建',
  queued: '排队中',
  dispatching: '提交中',
  running: '生成中',
  succeeded: '已完成',
  failed: '生成失败',
  canceled: '已取消',
}

export const RUNNING_STATUSES = [
  'created',
  'queued',
  'dispatching',
  'running',
]

export function isRunning(status: string): boolean {
  return RUNNING_STATUSES.includes(status)
}

export function isTerminal(status: string): boolean {
  return ['succeeded', 'failed', 'canceled'].includes(status)
}
```

## 二、TaskCard.vue 示例

```vue
<template>
  <div class="task-card">
    <div class="previews">
      <img :src="inputUrl || placeholderUrl" class="thumb" />
      <img v-if="outputUrl" :src="outputUrl" class="thumb" />
    </div>

    <div class="meta">
      <span>{{ task.toolName }}</span>
      <span>{{ STATUS_LABEL[task.status] || task.status }}</span>
    </div>

    <el-progress
      v-if="isRunning(task.status)"
      :percentage="task.progress"
      :stroke-width="6"
    />

    <p v-if="task.status === 'failed'" class="error">
      {{ task.errorMessage }}
    </p>

    <div class="actions">
      <el-button v-if="task.status === 'succeeded'" size="small" @click="emit('rerun')">
        再次生成
      </el-button>
      <el-button v-if="task.status === 'failed'" size="small" @click="emit('rerun')">
        再次生成
      </el-button>
      <el-button
        v-if="isRunning(task.status)"
        size="small"
        @click="emit('cancel')"
      >
        取消
      </el-button>
      <el-button size="small" type="danger" @click="emit('delete')">
        删除
      </el-button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import type { GenerationTask } from '@/types/task'
import { STATUS_LABEL, isRunning } from '@/utils/taskStatus'

const props = defineProps<{ task: GenerationTask }>()
const emit = defineEmits<{
  rerun: []
  cancel: []
  delete: []
}>()

const placeholderUrl = '/placeholder.png'

const inputUrl = computed(() => {
  const list = props.task.taskSubmitParams?.refImageList
  return Array.isArray(list) && list[0] ? String(list[0]) : ''
})

const outputUrl = computed(() => {
  const list = props.task.taskCallbackParams?.imageUrlList
  return Array.isArray(list) && list[0] ? String(list[0]) : ''
})
</script>
```

操作均通过 **HTTP** 触发，WS 只同步状态。见 [API接口.md](../API接口.md)。

## 三、taskStore 筛选

```ts
getters: {
  filteredTasks(state) {
    if (state.currentFilter === '进行中') {
      return state.tasks.filter(t => RUNNING_STATUSES.includes(t.status))
    }
    if (state.currentFilter === '已完成') {
      return state.tasks.filter(t => t.status === 'succeeded')
    }
    if (state.currentFilter === '失败') {
      return state.tasks.filter(t => t.status === 'failed')
    }
    if (state.currentFilter === '收藏') {
      return state.tasks.filter(t => t.favorite)
    }
    return state.tasks
  },
},
```

## 五、normalize_status（回调用）

```python
def normalize_status(raw: str | None) -> str:
    if raw in ("success", "succeeded", "completed"):
        return "succeeded"
    if raw in ("failed", "error"):
        return "failed"
    if raw in ("canceled", "cancelled"):
        return "canceled"
    return raw or "failed"
```

imggen MQ 回调仍用 `success`，映射为 `succeeded`。

## 六、create_task 完整流转（伪代码）

```python
async def create_task(user_id: str, data: dict):
    tool = get_tool(data["toolKey"])
    params = data.get("params") or {}
    tool.validate(params)

    task_id = generate_task_id()
    inner_task_id = generate_inner_task_id()
    dispatch = tool.build_task_dispatch_params(task_id, inner_task_id, params)

    task = save_task(
        user_id=user_id,
        task_id=task_id,
        inner_task_id=inner_task_id,
        status="created",
        progress=0,
        task_submit_params=params,
        task_dispatch_params=dispatch,
    )
    await ws.send_task_event(user_id, "task.created", task)

    task = update_status(task_id, "dispatching", 5)
    await ws.send_task_event(user_id, "task.updated", task)

    try:
        mq_service.send(dispatch, tool_key=tool.key)
        task = update_status(task_id, "queued", 10)
        await ws.send_task_event(user_id, "task.updated", task)
    except Exception:
        task = mark_failed(task_id, "任务提交失败，请稍后重试")
        await ws.send_task_event(user_id, "task.failed", task)

    return task
```

## 八、MQ 回调后更新

```python
def handle_imggen_callback(message: dict):
    inner_task_id = message["innerTaskId"]
    task = get_by_inner_task_id(inner_task_id)
    if not task or task.status in TERMINAL_STATUSES:
        return

    if message.get("status") in ("queued", "dispatching", "running"):
        progress = int(message.get("progress") or 50)
        updated = update_status(task.task_id, message["status"], progress)
        ws.send_task_event(task.user_id, "task.updated", updated)
        return

    if message.get("status") in ("success", "succeeded", "completed"):
        updated = mark_succeeded(
            task_id=task.task_id,
            task_callback_params=message,
            cost_time=message.get("costTime"),
        )
        ws.send_task_event(task.user_id, "task.succeeded", updated)
    else:
        updated = mark_failed(
            task_id=task.task_id,
            error_message=message.get("errorMsg") or "生成失败",
            task_callback_params=message,
        )
        ws.send_task_event(task.user_id, "task.failed", updated)
```

## 九、时间线示意

```text
T+0s   POST /tasks                 status=created     → WS task.created
T+0s   UcubMQDraw 投递 MQ           status=dispatching → WS task.updated
T+0s   MQ 投递成功                  status=queued      → WS task.updated
T+1s   imggen-stub 消费任务          可选 running       → WS task.updated
T+3s   imggen-stub 回调 success      status=succeeded  → WS task.succeeded
       taskCallbackParams.imageUrlList 有值 → 右栏显示输出图
```
